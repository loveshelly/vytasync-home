document.addEventListener('DOMContentLoaded', function () {
  const btnContainer = document.querySelector('.dw_box_button');
  const buttons = btnContainer ? Array.from(btnContainer.querySelectorAll('button')) : [];
  const offcanvas = document.querySelector('.dw_box_offcanvas');
  const closeBtn = document.querySelector('.dw_box_offcanvas_close');
  const carouselEl = document.getElementById('dw_box');
  let backdrop = null;

  const setActive = (idx) => {
    buttons.forEach(b => b.classList.remove('active'));
    const target = btnContainer?.querySelector(`[data-bs-slide-to="${idx}"]`) || buttons[idx];
    if (target) target.classList.add('active');
  };

  const lockScroll = (locked) => {
    const v = locked ? 'hidden' : '';
    document.documentElement.style.overflow = v;
    document.body.style.overflow = v;
  };

  const ensureBackdrop = () => {
    if (backdrop) return backdrop;
    backdrop = document.createElement('div');
    backdrop.className = 'dw_offcanvas_backdrop';
    Object.assign(backdrop.style, {
      position: 'fixed', top: '0', left: '0', right: '0', bottom: '0',
      background: 'rgba(0,0,0,0.4)',
      zIndex: '8'
    });
    backdrop.addEventListener('click', hide);
    backdrop.addEventListener('touchstart', hide, { passive: true });
    document.body.appendChild(backdrop);
    return backdrop;
  };

  const show = () => {
    if (!offcanvas) return;
    offcanvas.classList.add('shows');
    lockScroll(true);
    ensureBackdrop();
  };

  const hide = () => {
    if (!offcanvas) return;
    offcanvas.classList.remove('shows');
    buttons.forEach(b => b.classList.remove('active'));
    lockScroll(false);
    if (backdrop?.parentNode) {
      backdrop.parentNode.removeChild(backdrop);
      backdrop = null;
    }
  };

  buttons.forEach(btn => btn.addEventListener('click', () => {
    buttons.forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    show();
  }));

  if (closeBtn) closeBtn.addEventListener('click', hide);

  if (carouselEl) {
    const items = Array.from(carouselEl.querySelectorAll('.carousel-item'));
    carouselEl.addEventListener('slid.bs.carousel', (e) => {
      if (offcanvas && offcanvas.classList.contains('shows')) {
          const idx = typeof e.to === 'number' ? e.to : items.findIndex(i => i.classList.contains('active'));
          setActive(idx);
      }
    });
  }
});
